📌 Задача

Создать диалоговый интерфейс для общения с ИИ-моделью через OpenRouter API. Чат должен поддерживать контекст диалога, выбор модели и работать в реальном времени.

🔧 Решение

Использовал OpenRouter API — сервис, который предоставляет унифицированный доступ к множеству моделей (Llama, Mistral, Gemini и др.) через один API-ключ.

Основные шаги:

1. Получение API-ключа на openrouter.ai
2. Создание интерфейса чата (HTML + CSS)
3. Написание JavaScript-логики для отправки запросов и отображения ответов
4. Добавление поддержки контекста (история диалога)
5. Выбор модели через выпадающий список

📄 Код: Подключение к OpenRouter ▼ Показать код 
// ===== КОНФИГУРАЦИЯ =====
const OPENROUTER_API_KEY = 'sk-or-v1-...';
const OPENROUTER_URL = 'https://openrouter.ai/api/v1/chat/completions';

let conversationHistory = [];
let isProcessing = false;

// ===== ДОБАВЛЕНИЕ СООБЩЕНИЯ =====
function addMessage(role, content) {
    const msgDiv = document.createElement('div');
    msgDiv.className = `chat-message ${role}`;
    msgDiv.innerHTML = `
        <div class="message-role">${role === 'user' ? '👤 Вы' : '🤖 Ассистент'}</div>
        ${content}
        <span class="message-time">${new Date().toLocaleTimeString()}</span>
    `;
    chatMessages.appendChild(msgDiv);
    chatMessages.scrollTop = chatMessages.scrollHeight;
}
📄 Код: Отправка запроса и получение ответа ▼ Показать код 
// ===== ОТПРАВКА ЗАПРОСА =====
async function sendToAI(userMessage) {
    if (isProcessing) return;

    isProcessing = true;
    conversationHistory.push({ role: 'user', content: userMessage });
    addMessage('user', userMessage);

    try {
        const response = await fetch(OPENROUTER_URL, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${OPENROUTER_API_KEY}`,
                'HTTP-Referer': window.location.href,
                'X-Title': 'Slip_Star Chat'
            },
            body: JSON.stringify({
                model: modelSelect.value,
                messages: conversationHistory,
                max_tokens: 1000,
                temperature: 0.7,
                stream: false
            })
        });

        const data = await response.json();
        const assistantMessage = data.choices[0].message.content;

        conversationHistory.push({ role: 'assistant', content: assistantMessage });
        addMessage('assistant', assistantMessage);

    } catch (error) {
        addMessage('assistant', `⚠️ Ошибка: ${error.message}`);
    } finally {
        isProcessing = false;
        messageInput.focus();
    }
}
📄 Код: Автоматический fallback ▼ Показать код 
// ===== АВТОМАТИЧЕСКИЙ FALLBACK =====
if (errorMsg.includes('No endpoints found') || 
    errorMsg.includes('Insufficient credits')) {
    
    if (modelSelect.value !== 'openrouter/free') {
        statusIndicator.textContent = '⏳ Модель недоступна, пробую OpenRouter Free...';
        modelSelect.value = 'openrouter/free';
        
        const retryResponse = await fetch(OPENROUTER_URL, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${OPENROUTER_API_KEY}`
            },
            body: JSON.stringify({
                model: 'openrouter/free',
                messages: conversationHistory,
                max_tokens: 1000,
                temperature: 0.7,
                stream: false
            })
        });
        
        const retryData = await retryResponse.json();
        const assistantMessage = retryData.choices[0].message.content;
        addMessage('assistant', assistantMessage);
    }
}
